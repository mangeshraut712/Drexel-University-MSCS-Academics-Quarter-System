JMP start		; Jump to start Labeltext:  			; Text label	DB "Apple"	; Variable word Apple       	DB 0		; String terminatorstart:			; Start Label	MOV C, text   	; Point to var move text in C	MOV D, 236	; Point to output move the 236 in Dloop:			; Loop label	MOV A, [C]	; Get char from variable C	MOV [D], A	; Write to output from A	INC C		; Increment C char by 1	DEC D  		; Decrement D position by 1	CMP D, 231	; Compare the stack pointer value which is 231	JNZ loop		; jump if not zero to loop label	JMP exit		; jump to exit labelexit: 			; exit label	HLT		; stop the process
