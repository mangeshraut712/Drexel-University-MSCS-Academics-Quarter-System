JMP start			; jump to start labelx: DB 10				; label x should be the first number example int 10y: DB 17				; label y should be the last number example int 17z: DB 0				; label z is the result example 0start:				; label start 	MOV A, [x]		; move the value of x to A	MOV B, [y]		; move the value of y to B	MOV C, A			; move the value of A to C	JMP loop			; jump to loop labelloop:				; label loop	INC A			; increment A by 1	ADD C, A			; Addition of A and C A=A+C	CMP B, A			; Campare A is equal to B	JNZ loop			; jump if no zero to loop label	JMP exit			; jump to exit labelexit:				; label exit	MOV [z], C		; move the result to z	HLT			; stop the process
