JMP start			; jump to start labelx: DB 3				; label x should be the base number example int 3y: DB 4				; label y should be the power example int 4z: DB 0				; label z is the result example int 0start:				; start label	MOV A, 1			; var A is int = 1	MOV B, [y]		; var B int of [y] value	MOV C, [x]		; var C int of [x] value	JMP loop			; jump to loop labelloop:				;Loop to find the power of number	MUL C			;multiply C value	DEC B			;Decrement Value of B by 1	CMP B, 0			;Compare Value of B till 0	JNZ loop			;Jump if there is no zero to loop label	JMP exit			;jump to exit label exit:				;exit label to stop process	MOV [z], A		;var [z] int = A	HLT			;stop the process
